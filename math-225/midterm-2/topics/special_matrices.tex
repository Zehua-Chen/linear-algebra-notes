\section{Special Matrices}

  \subsection{Identity Matrix}
  
    \begin{equation}
      \begin{bmatrix}
        1 & 0 & 0 \\
        0 & 1 & 0 \\ 
        0 & 0 & 1
      \end{bmatrix}
    \end{equation}
    
    \begin{itemize}
      \item All \highlight{non zero cells has to be $ 0 $};
    \end{itemize}
    
    \subsubsection{Properties}
    
      \begin{enumerate}
        \item \highlight{An identity matrix is invertible};
      \end{enumerate}
      
  \subsubsection{Diagnoal Matrix}
  
    \begin{equation}
      \begin{bmatrix}
        2 & 0 & 0 \\
        0 & 1 & 0 \\
        0 & 0 & -2
      \end{bmatrix}
    \end{equation}
    
    \begin{itemize}
      \item \highlight{Non zero cells can be of any value};
    \end{itemize}
    
  \subsection{Triangular Matrix}
  
    \begin{equation}
      \begin{bmatrix}
        2 & 1 & 3 \\
        0 & 3 & -2 \\ 
        0 & 0 & 1
      \end{bmatrix}
    \end{equation}
    
    \begin{itemize}
      \item A \highlight{triangular matrix is a diagnoal matrix 
      with only half of its cells having values of $ 0 $};
    \end{itemize}
