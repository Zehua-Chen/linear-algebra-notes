\section{Elementary Matrices}

\begin{definition}
  $ E $ is an elementary matrix such that 
  \begin{displaymath}
    E \cdot A = \textbf{performing row operation on } A
  \end{displaymath}
\end{definition}

\subsection{Properties}

  \begin{enumerate}
    \item Each \highlight{elementary matrix is invertible};
    \begin{itemize}
      \item Each elementary matrix is a row operation, 
      \highlight{and every row operation is inversible};
      \item Therefore, each elementary matrix is invertible;
    \end{itemize}
  \end{enumerate}
