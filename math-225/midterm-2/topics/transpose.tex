\section{Transpose}

\begin{definition}
  A transpose is to \highlight{take every cell in a matrix and swap their
  $ i $ and $ j $ value}.
  \begin{displaymath}
    A^{T}_{ij} = A_{ji}
  \end{displaymath}
\end{definition}

\subsection{Properties}
  
  \begin{align*}
    \left( A + B \right)^{T} &= A^{T} + B^{T} \\
    \left( AB \right)^{T} &= B^{T} A^{T}
  \end{align*}
