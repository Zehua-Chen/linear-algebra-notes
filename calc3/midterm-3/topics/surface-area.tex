\section{Surface Area}

  \begin{align}
    SA
      &= \int \int_{S} dS \\
    dS
      &= \left| \vec{r}_{u} \times \vec{r}_{v} \right| du dv \\
      &= \sqrt{1 + \left( \frac{dz}{dx} \right)^{2} + \left( \frac{dz}{dy} \right)^{2}} dx dy \\
      &= \sqrt{1 + \left( \frac{dy}{dz} \right)^{2} + \left( \frac{dy}{dx} \right)^{2}} dx dz
  \end{align}
