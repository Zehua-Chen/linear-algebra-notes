\section{Area}

  \begin{equation}
    \int \left( \int dy \right) dx
  \end{equation}

  \subsection{Polar Coordinates}

    \begin{equation}
      dA = r dr d\theta
    \end{equation}

  \subsection{With Transform}

    \begin{equation}
      \int \int_{R} f\left( x, y \right) dA
        = \int \int_{S} f\left( T\left( u, v \right) \right) \left| \det J \right| du dv
    \end{equation}
