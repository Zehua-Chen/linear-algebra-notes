\section{Derivative}

  \subsection{Chain Rule}

    Given $ h(t) $
    \begin{displaymath}
      h(t) = f(x(t), y(t))
    \end{displaymath}

    \begin{align}
      h'(t)
        &= f_{x}\left( x(t), y(t) \right) x'(t)
        + f_{y} \left(x(t), y(t) \right) y'(t) \\
      \frac{dh}{dt}
        &= \frac{df}{dx} \frac{dx}{dt}
        + \frac{df}{dy} \frac{dy}{dt}
    \end{align}

    For more dimensions
    \begin{equation}
      \frac{dh}{dt}
        = \frac{df}{dx} \frac{dx}{dt}
        + \frac{df}{dy} \frac{dy}{dt}
        + \frac{df}{dz} \frac{dz}{dt}
    \end{equation}

  \subsection{Colaence Theorem}
