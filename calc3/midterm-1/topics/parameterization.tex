\section{Parameterization}

  Given two points $ \left( a, b, c \right) $ and $ \left( d, e, f \right) $

  \begin{align}
    \left(
      x\left( t \right), y\left( t \right), z\left( t \right)
    \right)
    &=
    \left(
      \left( d - a \right)t + a,
      \left( e - b \right)t + b,
      \left( f - c \right)t + c
    \right) \\
    f\left( t \right) &=
    t \left( d - a, e - b, f - c \right)
    + \left( a, b, c  \right)
  \end{align}
