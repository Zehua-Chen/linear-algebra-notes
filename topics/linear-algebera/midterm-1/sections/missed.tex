\begin{figure}[h!]\label{missed sample augmented matrix}
  \caption{Augmented Matrix}
  \begin{displaymath}
    \begin{bmatrix}

    \end{bmatrix}

\begin{figure}[h!]\label{missed sample matrix equation}
  \caption{Matrix Equation}

      a_{11} & a_{12} \\ 
      a_{21} & a_{22} \\ 
      a_{31} & a_{32}
    \end{bmatrix}
    \times 
    \begin{bmatrix}
      x_{1} \\ 
      x_{2} 
    \end{bmatrix}
    =
    \begin{bmatrix}
      b_{1} \\ 
      b_{2} \\ 
      b_{3}
    \end{bmatrix}
  \end{displaymath}
\end{figure}
